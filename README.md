        case 1: // С погрешностью eps
            while (true) {
                cout << "Введите eps (0 < eps < 1): ";
                getline(cin, input);
                stringstream ss(input);
                if (ss >> eps && ss.eof() && eps > 0 && eps < 1) {
                    break;
                }
                cout << "Ошибка: eps должен быть числом 0 < eps < 1" << endl;
            }
            y = 0;
            double u = 1;
            int k = 0;
            while (fabs(u) > eps) {
                y += u;
                k++;
                u = u * x / k;
            }
            break;

        case 2: // Первые k членов
            while (true) {
                cout << "Введите k (>= 0): ";
                getline(cin, input);
                bool valid = true;
                k = 0;
                for (char c : input) {
                    if (c < '0' || c > '9') {
                        valid = false;
                        break;
                    }
                    k = k * 10 + (c - '0');
                }
                if (valid && k >= 0) break;
                cout << "Ошибка: k должен быть целым числом >= 0" << endl;
            }
            y = 0;
            u = 1;
            for (int i = 0; i < k; i++) {
                y += u;
                u = u * x / (i + 1);
            }
            break;

        case 3: // Аналитическая формула
            y = exp(x);
            break;
    }

    // Вывод результата в разных форматах
    cout << fixed << setprecision(6) << "Результат (фиксированный формат): " << y << endl;
    cout << scientific << setprecision(6) << "Результат (научный формат): " << y << endl;

    return 0;
}
